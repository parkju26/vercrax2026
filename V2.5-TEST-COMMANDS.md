# Vercrax v2.5 테스트 (Windows / PowerShell)

## 준비
cd D:\AGENT\VERCRAX
$debug="d296c55f-beea-429e-ac58-5f12a19d12b3"

## Body 파일 생성
@'
{
  "prompt": "비트코인을 지금 투자할까요? 내 현금 1억, 12개월, 손실은 크게 못봄.",
  "mode": "deep",
  "debate": "all",
  "provider_preference": null,
  "stream": true
}
'@ | Out-File .\body.json -Encoding utf8

## 스트리밍 테스트 (NDJSON) - curl.exe 권장
curl.exe -N -s `
  -H "Content-Type: application/json" `
  --data-binary "@body.json" `
  "http://localhost:3000/api/judge?debug_user_id=$debug"

## 비스트리밍 테스트 (JSON 한방) - irm 사용
$body = @{
  prompt = "비트코인을 지금 투자할까요? 내 현금 1억, 12개월, 손실은 크게 못봄."
  mode = "deep"
  debate = "all"
  stream = $false
} | ConvertTo-Json -Depth 10

irm "http://localhost:3000/api/judge?debug_user_id=$debug" -Method Post -ContentType "application/json" -Body $body
